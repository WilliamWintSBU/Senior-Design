66HAL_StatusTypeDef ADS1115_Init(I2C_HandleTypeDef *handler, uint16_t setDataRate, uint16_t setPGA) {
67
68 // Handler
69 memcpy(&ADS1115_I2C_Handler, handler, sizeof(*handler));
70
71 // Data rate and PGA configurations.
72 ADS1115_dataRate = setDataRate;
73 ADS1115_pga = setPGA;
74
75 // Voltage coefficient update.
76 switch (ADS1115_pga) {
77
78 case ADS1115_PGA_TWOTHIRDS:
79 ADS1115_voltCoef = 0.1875;
80 break;
81
82 case ADS1115_PGA_ONE:
83 ADS1115_voltCoef = 0.125;
84 break;
85
86 case ADS1115_PGA_TWO:
87 ADS1115_voltCoef = 0.0625;
88 break;
89
90 case ADS1115_PGA_FOUR:
91 ADS1115_voltCoef = 0.03125;
92 break;
93
94 case ADS1115_PGA_EIGHT:
95 ADS1115_voltCoef = 0.015625;
96 break;
97
98 case ADS1115_PGA_SIXTEEN:
99 ADS1115_voltCoef = 0.0078125;
100 break;
101
102 }
103
104 if (HAL_I2C_IsDeviceReady(&ADS1115_I2C_Handler, (uint16_t) (ADS1115_devAddress << 1), 5, ADS1115_TIMEOUT) == HAL_OK) {
105 return HAL_OK;
106 } else {
107 return HAL_ERROR;
108 }
109
110}

112HAL_StatusTypeDef ADS1115_readSingleEnded(uint16_t muxPort, float *voltage) {
113
114 ADS1115_config[0] = ADS1115_OS | muxPort | ADS1115_pga | ADS1115_MODE;
115 ADS1115_config[1] = ADS1115_dataRate | ADS1115_COMP_MODE | ADS1115_COMP_POL | ADS1115_COMP_LAT| ADS1115_COMP_QUE;
116
117 if(HAL_I2C_Mem_Write(&ADS1115_I2C_Handler, (uint16_t) (ADS1115_devAddress << 1), ADS1115_CONFIG_REG, 1, ADS1115_config, 2, ADS1115_TIMEOUT) == HAL_OK){
118
119 if(HAL_I2C_Mem_Read(&ADS1115_I2C_Handler, (uint16_t) ((ADS1115_devAddress << 1) | 0x1), ADS1115_CONVER_REG, 1, ADS1115_rawValue, 2, ADS1115_TIMEOUT) == HAL_OK){
120
121 *voltage = (float) (((int16_t) (ADS1115_rawValue[0] << 8) | ADS1115_rawValue[1]) * ADS1115_voltCoef);
122 return HAL_OK;
123
124 }
125
126 }
127
128 return HAL_ERROR;
129
130}
