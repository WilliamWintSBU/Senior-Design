78static int32_t
79prv_parse_number(lwgps_t* gh, const char* t) {
80 int32_t res = 0;
81 uint8_t minus;
82
83 if (t == NULL) {
84 t = gh->p.term_str;
85 }
86 for (; t != NULL && *t == ' '; ++t) {} /* Strip leading spaces */
87
88 minus = (*t == '-' ? (++t, 1) : 0);
89 for (; t != NULL && CIN(*t); ++t) {
90 res = 10 * res + CTN(*t);
91 }
92 return minus ? -res : res;
93}

101static lwgps_float_t
102prv_parse_float_number(lwgps_t* gh, const char* t) {
103 lwgps_float_t value = (lwgps_float_t)0, power = (lwgps_float_t)1;
104 int sign = 1;
105
106 if (t == NULL) {
107 t = gh->p.term_str;
108 }
109 for (; t != NULL && *t == ' '; ++t) {} /* Strip leading spaces */
110
111 if (*t == '-') { /* Check sign */
112 sign = -1;
113 ++t;
114 }
115 while (CIN(*t)) { /* Convert main part */
116 value = value * (lwgps_float_t)10 + CTN(*t);
117 ++t;
118 }
119 if (*t == '.') { /* Skip the dot character */
120 ++t;
121 }
122 while (CIN(*t)) { /* Get the power */
123 value = value * (lwgps_float_t)10 + CTN(*t);
124 power *= 10.0;
125 ++t;
126 }
127 return sign * value / power;
128}

137static lwgps_float_t
138prv_parse_lat_long(lwgps_t* gh) {
139 lwgps_float_t ll, deg, min;
140
141 ll = prv_parse_float_number(gh, NULL); /* Parse value as double */
142 deg = FLT((int)((int)ll / 100)); /* Get absolute degrees value, interested in integer part only */
143 min = ll - (deg * FLT(100)); /* Get remaining part from full number, minutes */
144 ll = deg + (min / FLT(60.0)); /* Calculate latitude/longitude */
145
146 return ll;
147}

154static uint8_t
155prv_parse_term(lwgps_t* gh) {
156 if (gh->p.term_num == 0) { /* Check string type */
157 if (0) {
158#if LWGPS_CFG_STATEMENT_GPGGA
159 } else if (!strncmp(gh->p.term_str, "$GPGGA", 6) || !strncmp(gh->p.term_str, "$GNGGA", 6)) {
160 gh->p.stat = STAT_GGA;
161#endif /* LWGPS_CFG_STATEMENT_GPGGA */
162#if LWGPS_CFG_STATEMENT_GPGSA
163 } else if (!strncmp(gh->p.term_str, "$GPGSA", 6) || !strncmp(gh->p.term_str, "$GNGSA", 6)) {
164 gh->p.stat = STAT_GSA;
165#endif /* LWGPS_CFG_STATEMENT_GPGSA */
166#if LWGPS_CFG_STATEMENT_GPGSV
167 } else if (!strncmp(gh->p.term_str, "$GPGSV", 6) || !strncmp(gh->p.term_str, "$GNGSV", 6)) {
168 gh->p.stat = STAT_GSV;
169#endif /* LWGPS_CFG_STATEMENT_GPGSV */
170#if LWGPS_CFG_STATEMENT_GPRMC
171 } else if (!strncmp(gh->p.term_str, "$GPRMC", 6) || !strncmp(gh->p.term_str, "$GNRMC", 6)) {
172 gh->p.stat = STAT_RMC;
173#endif /* LWGPS_CFG_STATEMENT_GPRMC */
174#if LWGPS_CFG_STATEMENT_PUBX
175 } else if (!strncmp(gh->p.term_str, "$PUBX", 5)) {
176 gh->p.stat = STAT_UBX;
177#endif /* LWGPS_CFG_STATEMENT_PUBX */
178 } else {
179 gh->p.stat = STAT_UNKNOWN; /* Invalid statement for library */
180 }
181 return 1;
182 }
183
184 /* Start parsing terms */
185 if (gh->p.stat == STAT_UNKNOWN) {
186#if LWGPS_CFG_STATEMENT_GPGGA
187 } else if (gh->p.stat == STAT_GGA) { /* Process GPGGA statement */
188 switch (gh->p.term_num) {
189 case 1: /* Process UTC time */
190 gh->p.data.gga.hours = 10 * CTN(gh->p.term_str[0]) + CTN(gh->p.term_str[1]);
191 gh->p.data.gga.minutes = 10 * CTN(gh->p.term_str[2]) + CTN(gh->p.term_str[3]);
192 gh->p.data.gga.seconds = 10 * CTN(gh->p.term_str[4]) + CTN(gh->p.term_str[5]);
193 break;
194 case 2: /* Latitude */
195 gh->p.data.gga.latitude = prv_parse_lat_long(gh); /* Parse latitude */
196 break;
197 case 3: /* Latitude north/south information */
198 if (gh->p.term_str[0] == 'S' || gh->p.term_str[0] == 's') {
199 gh->p.data.gga.latitude = -gh->p.data.gga.latitude;
200 }
201 break;
202 case 4: /* Longitude */
203 gh->p.data.gga.longitude = prv_parse_lat_long(gh); /* Parse longitude */
204 break;
205 case 5: /* Longitude east/west information */
206 if (gh->p.term_str[0] == 'W' || gh->p.term_str[0] == 'w') {
207 gh->p.data.gga.longitude = -gh->p.data.gga.longitude;
208 }
209 break;
210 case 6: /* Fix status */
211 gh->p.data.gga.fix = (uint8_t)prv_parse_number(gh, NULL);
212 break;
213 case 7: /* Satellites in use */
214 gh->p.data.gga.sats_in_use = (uint8_t)prv_parse_number(gh, NULL);
215 break;
216 case 9: /* Altitude */
217 gh->p.data.gga.altitude = prv_parse_float_number(gh, NULL);
218 break;
219 case 11: /* Altitude above ellipsoid */
220 gh->p.data.gga.geo_sep = prv_parse_float_number(gh, NULL);
221 break;
222 default:
223 break;
224 }
225#endif /* LWGPS_CFG_STATEMENT_GPGGA */
226#if LWGPS_CFG_STATEMENT_GPGSA
227 } else if (gh->p.stat == STAT_GSA) { /* Process GPGSA statement */
228 switch (gh->p.term_num) {
229 case 2: /* Process fix mode */
230 gh->p.data.gsa.fix_mode = (uint8_t)prv_parse_number(gh, NULL);
231 break;
232 case 15: /* Process PDOP */
233 gh->p.data.gsa.dop_p = prv_parse_float_number(gh, NULL);
234 break;
235 case 16: /* Process HDOP */
236 gh->p.data.gsa.dop_h = prv_parse_float_number(gh, NULL);
237 break;
238 case 17: /* Process VDOP */
239 gh->p.data.gsa.dop_v = prv_parse_float_number(gh, NULL);
240 break;
241 default:
242 /* Parse satellite IDs */
243 if (gh->p.term_num >= 3 && gh->p.term_num <= 14) {
244 gh->p.data.gsa.satellites_ids[gh->p.term_num - 3] = (uint8_t)prv_parse_number(gh, NULL);
245 }
246 break;
247 }
248#endif /* LWGPS_CFG_STATEMENT_GPGSA */
249#if LWGPS_CFG_STATEMENT_GPGSV
250 } else if (gh->p.stat == STAT_GSV) { /* Process GPGSV statement */
251 switch (gh->p.term_num) {
252 case 2: /* Current GPGSV statement number */
253 gh->p.data.gsv.stat_num = (uint8_t)prv_parse_number(gh, NULL);
254 break;
255 case 3: /* Process satellites in view */
256 gh->p.data.gsv.sats_in_view = (uint8_t)prv_parse_number(gh, NULL);
257 break;
258 default:
259#if LWGPS_CFG_STATEMENT_GPGSV_SAT_DET
260 if (gh->p.term_num >= 4 && gh->p.term_num <= 19) { /* Check current term number */
261 uint8_t index, term_num = gh->p.term_num - 4; /* Normalize term number from 4-19 to 0-15 */
262 uint16_t value;
263
264 index = ((gh->p.data.gsv.stat_num - 1) << 0x02) + (term_num >> 2); /* Get array index */
265 if (index < sizeof(gh->sats_in_view_desc) / sizeof(gh->sats_in_view_desc[0])) {
266 value = (uint16_t)prv_parse_number(gh, NULL); /* Parse number as integer */
267 switch (term_num & 0x03) {
268 case 0:
269 gh->sats_in_view_desc[index].num = value;
270 break;
271 case 1:
272 gh->sats_in_view_desc[index].elevation = value;
273 break;
274 case 2:
275 gh->sats_in_view_desc[index].azimuth = value;
276 break;
277 case 3:
278 gh->sats_in_view_desc[index].snr = value;
279 break;
280 default:
281 break;
282 }
283 }
284 }
285#endif /* LWGPS_CFG_STATEMENT_GPGSV_SAT_DET */
286 break;
287 }
288#endif /* LWGPS_CFG_STATEMENT_GPGSV */
289#if LWGPS_CFG_STATEMENT_GPRMC
290 } else if (gh->p.stat == STAT_RMC) { /* Process GPRMC statement */
291 switch (gh->p.term_num) {
292 case 2: /* Process valid status */
293 gh->p.data.rmc.is_valid = (gh->p.term_str[0] == 'A');
294 break;
295 case 7: /* Process ground speed in knots */
296 gh->p.data.rmc.speed = prv_parse_float_number(gh, NULL);
297 break;
298 case 8: /* Process true ground coarse */
299 gh->p.data.rmc.course = prv_parse_float_number(gh, NULL);
300 break;
301 case 9: /* Process date */
302 gh->p.data.rmc.date = (uint8_t)(10 * CTN(gh->p.term_str[0]) + CTN(gh->p.term_str[1]));
303 gh->p.data.rmc.month = (uint8_t)(10 * CTN(gh->p.term_str[2]) + CTN(gh->p.term_str[3]));
304 gh->p.data.rmc.year = (uint8_t)(10 * CTN(gh->p.term_str[4]) + CTN(gh->p.term_str[5]));
305 break;
306 case 10: /* Process magnetic variation */
307 gh->p.data.rmc.variation = prv_parse_float_number(gh, NULL);
308 break;
309 case 11: /* Process magnetic variation east/west */
310 if (gh->p.term_str[0] == 'W' || gh->p.term_str[0] == 'w') {
311 gh->p.data.rmc.variation = -gh->p.data.rmc.variation;
312 }
313 break;
314 default:
315 break;
316 }
317#endif /* LWGPS_CFG_STATEMENT_GPRMC */
318#if LWGPS_CFG_STATEMENT_PUBX
319 } else if (gh->p.stat == STAT_UBX) { /* Disambiguate generic PUBX statement */
320 if (gh->p.term_str[0] == '0' && gh->p.term_str[1] == '4') {
321 gh->p.stat = STAT_UBX_TIME;
322 }
323#if LWGPS_CFG_STATEMENT_PUBX_TIME
324 } else if (gh->p.stat == STAT_UBX_TIME) { /* Process PUBX (uBlox) TIME statement */
325 switch (gh->p.term_num) {
326 case 2: /* Process UTC time; ignore fractions of seconds */
327 gh->p.data.time.hours = 10 * CTN(gh->p.term_str[0]) + CTN(gh->p.term_str[1]);
328 gh->p.data.time.minutes = 10 * CTN(gh->p.term_str[2]) + CTN(gh->p.term_str[3]);
329 gh->p.data.time.seconds = 10 * CTN(gh->p.term_str[4]) + CTN(gh->p.term_str[5]);
330 break;
331 case 3: /* Process UTC date */
332 gh->p.data.time.date = 10 * CTN(gh->p.term_str[0]) + CTN(gh->p.term_str[1]);
333 gh->p.data.time.month = 10 * CTN(gh->p.term_str[2]) + CTN(gh->p.term_str[3]);
334 gh->p.data.time.year = 10 * CTN(gh->p.term_str[4]) + CTN(gh->p.term_str[5]);
335 break;
336 case 4: /* Process UTC TimeOfWeek */
337 gh->p.data.time.utc_tow = prv_parse_float_number(gh, NULL);
338 break;
339 case 5: /* Process UTC WeekNumber */
340 gh->p.data.time.utc_wk = prv_parse_number(gh, NULL);
341 break;
342 case 6: /* Process UTC leap seconds */
343 /*
344 * Accomodate a 2- or 3-digit leap second count
345 * a trailing 'D' means this is the firmware's default value.
346 */
347 if (gh->p.term_str[2] == 'D' || gh->p.term_str[2] == '\0') {
348 gh->p.data.time.leap_sec = 10 * CTN(gh->p.term_str[0]) + CTN(gh->p.term_str[1]);
349 } else {
350 gh->p.data.time.leap_sec =
351 100 * CTN(gh->p.term_str[0]) + 10 * CTN(gh->p.term_str[1]) + CTN(gh->p.term_str[2]);
352 }
353 break;
354 case 7: /* Process clock bias */
355 gh->p.data.time.clk_bias = prv_parse_number(gh, NULL);
356 break;
357 case 8: /* Process clock drift */
358 gh->p.data.time.clk_drift = prv_parse_float_number(gh, NULL);
359 break;
360 case 9: /* Process time pulse granularity */
361 gh->p.data.time.tp_gran = prv_parse_number(gh, NULL);
362 break;
363 default:
364 break;
365 }
366#endif /* LWGPS_CFG_STATEMENT_PUBX_TIME */
367#endif /* LWGPS_CFG_STATEMENT_PUBX */
368 }
369 return 1;
370}

379prv_check_crc(lwgps_t* gh) {
380 uint8_t crc;
381 crc = (uint8_t)((CHTN(gh->p.term_str[0]) & 0x0F) << 0x04)
382 | (CHTN(gh->p.term_str[1]) & 0x0F); /* Convert received CRC from string (hex) to number */

384}

470uint8_t
471#if LWGPS_CFG_STATUS || __DOXYGEN__
472lwgps_process(lwgps_t* gh, const void* data, size_t len, lwgps_process_fn evt_fn) {
473#else /* LWGPS_CFG_STATUS */
474lwgps_process(lwgps_t* gh, const void* data, size_t len) {
475#endif /* !LWGPS_CFG_STATUS */
476 const uint8_t* d = data;
477
478 for (; len > 0; ++d, --len) { /* Process all bytes */
479 if (*d == '$') { /* Check for beginning of NMEA line */
480 memset(&gh->p, 0x00, sizeof(gh->p)); /* Reset private memory */
481 TERM_ADD(gh, *d); /* Add character to term */
482 } else if (*d == ',') { /* Term separator character */
483 prv_parse_term(gh); /* Parse term we have currently in memory */
484 CRC_ADD(gh, *d); /* Add character to CRC computation */
485 TERM_NEXT(gh); /* Start with next term */
486 } else if (*d == '*') { /* Start indicates end of data for CRC computation */
487 prv_parse_term(gh); /* Parse term we have currently in memory */
488 gh->p.star = 1; /* STAR detected */
489 TERM_NEXT(gh); /* Start with next term */
490 } else if (*d == '\r') {
491 if (prv_check_crc(gh)) { /* Check for CRC result */
492 /* CRC is OK, in theory we can copy data from statements to user data */
493 prv_copy_from_tmp_memory(gh); /* Copy memory from temporary to user memory */
494#if LWGPS_CFG_STATUS
495 if (evt_fn != NULL) {
496 evt_fn(gh->p.stat);
497 }
498 } else if (evt_fn != NULL) {
499 evt_fn(STAT_CHECKSUM_FAIL);
500#endif /* LWGPS_CFG_STATUS */
501 }
502 } else {
503 if (!gh->p.star) { /* Add to CRC only if star not yet detected */
504 CRC_ADD(gh, *d); /* Add to CRC */
505 }
506 TERM_ADD(gh, *d); /* Add character to term */
507 }
508 }
509 return 1;
510}

524uint8_t
525lwgps_distance_bearing(lwgps_float_t las, lwgps_float_t los, lwgps_float_t lae, lwgps_float_t loe, lwgps_float_t* d,
526 lwgps_float_t* b) {
527 lwgps_float_t df, dfi, a;
528
529 if (d == NULL && b == NULL) {
530 return 0;
531 }
532
533 /* Convert degrees to radians */
534 df = D2R(lae - las);
535 dfi = D2R(loe - los);
536 las = D2R(las);
537 los = D2R(los);
538 lae = D2R(lae);
539 loe = D2R(loe);
540
541 /*
542 * Calculate distance
543 *
544 * Calculated distance is absolute value in meters between 2 points on earth.
545 */
546 if (d != NULL) {
547 /*
548 * a = sin(df / 2)^2 + cos(las) * cos(lae) * sin(dfi / 2)^2
549 * *d = RADIUS * 2 * atan(sqrt(a) / sqrt(1 - a)) * 1000 (for meters)
550 */
551#if LWGPS_CFG_DOUBLE
552 a = FLT(sin(df * 0.5) * sin(df * 0.5) + sin(dfi * 0.5) * sin(dfi * 0.5) * cos(las) * cos(lae));
553 *d = FLT(EARTH_RADIUS * 2.0 * atan2(sqrt(a), sqrt(1.0 - a)) * 1000.0);
554#else /* LWGPS_CFG_DOUBLE */
555 a = FLT(sinf(df * 0.5f) * sinf(df * 0.5f) + sinf(dfi * 0.5f) * sinf(dfi * 0.5f) * cosf(las) * cosf(lae));
556 *d = FLT(EARTH_RADIUS * 2.0f * atan2f(sqrtf(a), sqrtf(1.0f - a)) * 1000.0f);
557#endif /* !LWGPS_CFG_DOUBLE */
558 }
559
560 /*
561 * Calculate bearing
562 *
563 * Bearing is calculated from point 1 to point 2.
564 * Result will tell us in which direction (according to north) we should move,
565 * to reach point 2.
566 *
567 * Example:
568 * Bearing is 0 => move to north
569 * Bearing is 90 => move to east
570 * Bearing is 180 => move to south
571 * Bearing is 270 => move to west
572 */
573 if (b != NULL) {
574#if LWGPS_CFG_DOUBLE
575 df = FLT(sin(loe - los) * cos(lae));
