64void LCD_Delay_us(uint16_t us)
65{
66 uint32_t Div = (SysTick->LOAD+1)/1000;
67 uint32_t StartMicros = HAL_GetTick()*1000 + (1000- SysTick->VAL/Div);
68 while((HAL_GetTick()*1000 + (1000-SysTick->VAL/Div)-StartMicros < us));
69}

80void LCD_Init(void)
81{
82 GPIO_InitTypeDef gpio;
83 gpio.Mode = GPIO_MODE_OUTPUT_PP;
84 gpio.Speed = GPIO_SPEED_FREQ_HIGH;
85 gpio.Pull = GPIO_NOPULL;
86
87 gpio.Pin = _LCD_RS_PIN;
88 HAL_GPIO_Init(_LCD_RS_PORT,&gpio);
89 gpio.Pin = _LCD_E_PIN;
90 HAL_GPIO_Init(_LCD_E_PORT,&gpio);
91 gpio.Pin = _LCD_RW_PIN;
92 HAL_GPIO_Init(_LCD_RW_PORT,&gpio);
93 gpio.Pin = _LCD_D4_PIN;
94 HAL_GPIO_Init(_LCD_D4_PORT,&gpio);
95 gpio.Pin = _LCD_D5_PIN;
96 HAL_GPIO_Init(_LCD_D5_PORT,&gpio);
97 gpio.Pin = _LCD_D6_PIN;
98 HAL_GPIO_Init(_LCD_D6_PORT,&gpio);
99 gpio.Pin = _LCD_D7_PIN;
100 HAL_GPIO_Init(_LCD_D7_PORT,&gpio);
101
102 while(HAL_GetTick()<200)
103 LCD_Delay_ms(1);
104 /* Set cursor pointer to beginning for LCD */
105 LCD_Opts.currentX = 0;
106 LCD_Opts.currentY = 0;
107 LCD_Opts.DisplayFunction = LCD_4BITMODE | LCD_5x8DOTS | LCD_1LINE;
108 if (_LCD_ROWS > 1)
109 LCD_Opts.DisplayFunction |= LCD_2LINE;
110 /* Try to set 4bit mode */
111 LCD_Cmd4bit(0x03);
112 LCD_Delay_ms(5);
113 /* Second try */
114 LCD_Cmd4bit(0x03);
115 LCD_Delay_ms(5);
116 /* Third goo! */
117 LCD_Cmd4bit(0x03);
118 LCD_Delay_ms(5);
119 /* Set 4-bit interface */
120 LCD_Cmd4bit(0x02);
121 LCD_Delay_ms(5);
122 /* Set # lines, font size, etc. */
123 LCD_Cmd(LCD_FUNCTIONSET | LCD_Opts.DisplayFunction);
124 /* Turn the display on with no cursor or blinking default */
125 LCD_Opts.DisplayControl = LCD_DISPLAYON;
126 LCD_DisplayOn();
127 LCD_Clear();
128 /* Default font directions */
129 LCD_Opts.DisplayMode = LCD_ENTRYLEFT | LCD_ENTRYSHIFTDECREMENT;
130 LCD_Cmd(LCD_ENTRYMODESET | LCD_Opts.DisplayMode);
131 LCD_Delay_ms(5);
132}

140void LCD_Puts(uint8_t x, uint8_t y, char* str)
141{
142 LCD_CursorSet(x, y);
143 while (*str)
144 {
145 if (LCD_Opts.currentX >= _LCD_COLS)
146 {
147 LCD_Opts.currentX = 0;
148 LCD_Opts.currentY++;
149 LCD_CursorSet(LCD_Opts.currentX, LCD_Opts.currentY);
150 }
151 if (*str == '\n')
152 {
153 LCD_Opts.currentY++;
154 LCD_CursorSet(LCD_Opts.currentX, LCD_Opts.currentY);
155 }
156 else if (*str == '\r')
157 {
158 LCD_CursorSet(0, LCD_Opts.currentY);
159 }
160 else
161 {
162 LCD_Data(*str);
163 LCD_Opts.currentX++;
164 }
165 str++;
166 }
167}
