217{

219 if(ADS1115_read(ADS1115_MUX_AIN0_AIN3, &hall1.rawVoltage) == HAL_OK)
220 {
221 //Voltage in mV successfully read.

225 {
226 // if current read fails, set current to a conspicuous value

330void HAL_UART_RxCpltCallback(UART_HandleTypeDef *huart)
331{
332 if( huart == &huart3)
333 {
334 if(gps.buffStateFlag == write)
335 {
336 switch (gps.parsingState)
337 {
338 case init:
339 if(gps.newchar == '$')
340 {
341 gps.buff[gps.tail] = gps.newchar;
342 gps.tail++;
343 gps.parsingState = seen$;
344 }
345 break;
346 case seen$:
347 gps.buff[gps.tail] = gps.newchar;
348 gps.tail++;
349 if(gps.newchar == '\n')
350 {
351 gps.lineCounter++;
352 if (gps.lineCounter >= gps.numLines)
353 {
354 // full message complete
355 if(1) // check for time TKTKT
356 {
357 gps.buff[gps.tail + 1] = '0'; // null terminate buffer might not be necessary)
358 gps.buffStateFlag = read; // set flag for main program;
359 gps.parsingState = init;
360 gps.lineCounter = 0;
361 }
362 else
363 {
364 //too long since last line, restart
365 gps.tail = 0;
366 gps.parsingState = init;
367
368 }
369 }
370 }
371 break;
372 default : break;
373 }
374 }
375 HAL_UART_Receive_IT(&huart3 , (uint8_t *) & gps.newchar , 1);
376 }

427void HAL_TIM_IC_CaptureCallback(TIM_HandleTypeDef *htim){
428 if(htim->Channel == HAL_TIM_ACTIVE_CHANNEL_1) {
429 if(Is_First_Captured == 0){
430 IC_Val1 = HAL_TIM_ReadCapturedValue(htim, TIM_CHANNEL_1);
431 Is_First_Captured = 1;
432 }
433 else {
434 IC_Val2 = HAL_TIM_ReadCapturedValue(htim, TIM_CHANNEL_1);
435
436 if (IC_Val2 > IC_Val1){
437 Difference = IC_Val2-IC_Val1;
438 }
439 else if (IC_Val1 > IC_Val2){
440 Difference = (0xffffffff - IC_Val1) + IC_Val2;
441 }
442
443 float refClock = TIMCLOCK/(PRESCALAR);
444
445 Frequency = refClock/Difference;
446 rpm = Frequency * 60.0;
447
448 __HAL_TIM_SET_COUNTER(htim, 0);
449 Is_First_Captured = 0;
450 }
451 }

496float readCurrent(void){
497
498 HAL_ADC_PollForConversion(&hadc,1000);
499 readValue = HAL_ADC_GetValue(&hadc);
500 rawVoltage = (float) readValue * 3.3 * .9104151493 / 4095.0 ;
501 // If rawVoltage is not 2.5Volt, multiply by a factor.In my case it is 1.035
502 // This is due to tolerance in voltage divider resister & ADC accuracy
503 current =(rawVoltage - 2.4988276)/sensitivity;
504 return current;
