221 * using an ADS1115 external ADC. All 3 Vref outputs are connected

225{
226 // hall sensor 1
227 if(ADS1115_read(ADS1115_MUX_AIN0_AIN3, &hall1.rawVoltage) == HAL_OK)
228 {
229 //Voltage in mV successfully read.
230 hall1.current = (hall1.rawVoltage - hall1.offsetVoltage) * HALL_SENSITIVITY;

233 {
234 // if current read fails, set current to a conspicuous value
235 hall1.current = 123.456789;
236 }
237
238 // hall sensor 2

338void HAL_UART_RxCpltCallback(UART_HandleTypeDef *huart)
339{
340 if( huart == &huart3)
341 {
342 if(gps.buffStateFlag == write)
343 {
344 switch (gps.parsingState)
345 {
346 case init:
347 if(gps.newchar == '$')
348 {
349 gps.buff[gps.tail] = gps.newchar;
350 gps.tail++;
351 gps.parsingState = seen$;
352 }
353 break;
354 case seen$:
355 gps.buff[gps.tail] = gps.newchar;
356 gps.tail++;
357 if(gps.newchar == '\n')
358 {
359 gps.lineCounter++;
360 if (gps.lineCounter >= gps.numLines)
361 {
362 // full message complete
363 if(1) // check for time TKTKT
364 {
365 gps.buff[gps.tail + 1] = '0'; // null terminate buffer might not be necessary)
366 gps.buffStateFlag = read; // set flag for main program;
367 gps.parsingState = init;
368 gps.lineCounter = 0;
369 }
370 else
371 {
372 //too long since last line, restart
373 gps.tail = 0;
374 gps.parsingState = init;
375
376 }
377 }
378 }
379 break;
380 default : break;
381 }
382 }
383 HAL_UART_Receive_IT(&huart3 , (uint8_t *) & gps.newchar , 1);
384 }

435void HAL_TIM_IC_CaptureCallback(TIM_HandleTypeDef *htim){
436 if(htim->Channel == HAL_TIM_ACTIVE_CHANNEL_1) {
437 if(Is_First_Captured == 0){
438 IC_Val1 = HAL_TIM_ReadCapturedValue(htim, TIM_CHANNEL_1);
439 Is_First_Captured = 1;
440 }
441 else {
442 IC_Val2 = HAL_TIM_ReadCapturedValue(htim, TIM_CHANNEL_1);
443
444 if (IC_Val2 > IC_Val1){
445 Difference = IC_Val2-IC_Val1;
446 }
447 else if (IC_Val1 > IC_Val2){
448 Difference = (0xffffffff - IC_Val1) + IC_Val2;
449 }
450
451 float refClock = TIMCLOCK/(PRESCALAR);
452
453 Frequency = refClock/Difference;
454 rpm = Frequency * 60.0;
455
456 __HAL_TIM_SET_COUNTER(htim, 0);
457 Is_First_Captured = 0;
458 }
459 }

511float readCurrent(void){
512
513 HAL_ADC_PollForConversion(&hadc,1000);
514 readValue = HAL_ADC_GetValue(&hadc);
515 rawVoltage = (float) readValue * 3.3 * .9104151493 / 4095.0 ;
516 // If rawVoltage is not 2.5Volt, multiply by a factor.In my case it is 1.035
517 // This is due to tolerance in voltage divider resister & ADC accuracy
518 current =(rawVoltage - 2.4988276)/sensitivity;
519 return current;
