32void pcf8574_send_cmd(uint8_t cmd)
33{
34 // EXAMPLE
35 // cmd = 1101 1010 = 0xDA
36 // du = 1101 0000 = 0xD0
37 // dl = 1010 0000 = 0xA0
38 // d[0]= 1101 1100 = 0xDC
39 // d[1]= 1101 1000 = 0xD8
40 // d[2]= 1010 1100 = 0xAC
41 // d[3]= 1010 1000 = 0xA8
42
43 uint8_t du, dl;
44 uint8_t data[4];
45 du = cmd & 0xf0; // & 0b1111 0000
46 dl = (cmd << 4) & 0xf0;
47 data[0] = du | 0x0C; // | 0b0000 1100
48 data[1] = du | 0x08; // | 0b0000 1000
49 data[2] = dl | 0x0C; // | 0b0000 1100
50 data[3] = dl | 0x08; // | 0b0000 1000
51 HAL_I2C_Master_Transmit(&PCF8574_I2C_HANDLER, PCF8574_I2C_ADDRESS, data, sizeof(data), 0xf);
52}

54void pcf8574_send_data(uint8_t cmd)
55{
56 uint8_t du, dl;
57 uint8_t data[4];
58 du = cmd & 0xf0;
59 dl = (cmd << 4) & 0xf0;
60 data[0] = du | 0x0D; // | 0b0000 1101
61 data[1] = du | 0x09; // | 0b0000 1001
62 data[2] = dl | 0x0D; // | 0b0000 1101
63 data[3] = dl | 0x09; // | 0b0000 1001
64 HAL_I2C_Master_Transmit(&PCF8574_I2C_HANDLER, PCF8574_I2C_ADDRESS, data, sizeof(data), 0xf);
65}

76void pcf8574_cursor(uint8_t row, uint8_t col)
77{
78 switch (row)
79 {
80 case 0:
81 col |= 0x80;
82 break;
83 case 1:
84 col |= 0x80 | 0x40;
85 break;
86 case 2:
87 col |= 0x80 | 0x14;
88 break;
89 case 3:
90 col |= 0x80 | 0x54;
91 break;
92 }
93 pcf8574_send_cmd(col);
94}

135void pcf8574_init()
136{
137 for (uint8_t i = 0; i < 3; i++)
138 {
139 pcf8574_send_cmd(0x03); // init sequence
140 HAL_Delay(5);
141 }
142 pcf8574_send_cmd(0x02); // 4-bit mode
143 HAL_Delay(100);
144 pcf8574_send_cmd(0x28); // 2 rows, 5x7 points
145 HAL_Delay(1);
146 pcf8574_send_cmd(0x01); // clear
147 HAL_Delay(2);
